CÓMO USAR:
================================================================================

BACKEND (Express.js):
---------------------
const { requireAuth, requirePermission } = require('./backend/middleware/rbac-middleware');

app.post('/api/ventas',
  requireAuth,
  requirePermission('venta', 'crear'),
  async (req, res) => {
    // Crear venta...
  }
);

FRONTEND (HTML):
----------------
<!-- Atributo declarativo -->
<button
  data-rbac-recurso="venta"
  data-rbac-accion="crear"
  id="btn-crear">
  Crear Venta
</button>

<!-- Inicializar -->
<script src="./frontend/rbac/permissions.js"></script>
<script>
  PermissionSystem.initialize(usuarioActual);
  PermissionSystem.applyRBACToPage();
</script>

FRONTEND (JavaScript):
----------------------
if (PermissionSystem.hasPermission('venta', 'crear')) {
  // Mostrar botón
}

PermissionSystem.withPermissionCheck(
  'venta', 'crear',
  () => crearVenta(),
  'No tiene permiso'
);
